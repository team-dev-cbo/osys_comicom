IMPORT FICHIER ASCII, Mot Clé

 

With stdlib, stdstrin, databaseg;

Procedure ImportASCII;

 

Var

 File : T_Fichier;

 ChaineLue : TresGrandeChaine;

 Val1 : GrandeChaine;

 Val2 : GrandeChaine;

 ReponseBD : T_ReponseSQL;

 Requete : TresGrandeChaine;

 NumCanal : T_CanalBD;

 

Begin

 //Ouverture du fichier ASCII

 File := OuvreFichierLecture("C:\Import\Article.txt");

 

 //Réservation d'un canal pour accéder à la base de données

 NumCanal:=ReserveCanal;

 IF NumCanal.IdCanal =0 THEN

// Pas de réservation d'un canal

FermeFichier (File); //Ferme le fichier ASCII

  STOP; //Arrête l'exécution du programme

 END IF;

 

 //Vérifie si l'ouverture s'est bien passée

 IF File.PointeurFichier <> 0 THEN

ChaineLue := LectureFichier (File);

  WHILE (ChaineLue<>EOF) DO

//Traitement de la chaine lue

   Val1 := ExtraitChaineNumDel (ChaineLue,1,':');

   Val2 := ExtraitChaineNumDel (ChaineLue,2,':');

 

   //Ecriture dans la base de données

   Requete := "INSERT INTO Article (Champ1, Champ2) VALUES (" + Strq(Val1) + "," + Strq(Val2) + ")";

   // Exécute la requête sur le canal réservé

   ReponseBD :=ExecuteRequeteSQLC(Requete,NumCanal);

   //Lecture de la ligne suivante 

ChaineLue := LectureFichier (File);

  END WHILE;

  //Fermeture du fichier

FermeFichier (File);

 // Libère le canal

   LibereCanal(NumCanal);

 END IF;

End ImportASCII;
